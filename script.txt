-- Troque API_BASE pela URL publicada do seu Cloudflare Worker.

local API_BASE = "https://gag2-key-api.joao-gag2.workers.dev"
local KEY_FILE = "gag2_saved_key.txt"
local DEVICE_FILE = "gag2_installation_id.txt"

local HttpService = game:GetService("HttpService")

local ENV = _G
if type(getgenv) == "function" then
    local ok, result = pcall(getgenv)
    if ok and type(result) == "table" then
        ENV = result
    end
end

local function findRequestFunction()
    local candidates = {
        ENV.request,
        ENV.http_request,
        ENV.httprequest,
        _G.request,
        _G.http_request,
    }

    if type(syn) == "table" then
        table.insert(candidates, syn.request)
    end

    for _, candidate in ipairs(candidates) do
        if type(candidate) == "function" then
            return candidate
        end
    end

    return nil
end

local requestFunction = findRequestFunction()
if not requestFunction then
    error("Este ambiente nao disponibiliza request/http_request")
end

local function trim(value)
    return (tostring(value or ""):gsub("^%s+", ""):gsub("%s+$", ""))
end

local function readLocalFile(path)
    if type(readfile) ~= "function" then
        return nil
    end
    if type(isfile) == "function" then
        local fileOk, exists = pcall(isfile, path)
        if not fileOk or not exists then
            return nil
        end
    end
    local ok, value = pcall(readfile, path)
    if ok and type(value) == "string" then
        return value
    end
    return nil
end

local function writeLocalFile(path, value)
    if type(writefile) ~= "function" then
        return false
    end
    return pcall(writefile, path, tostring(value))
end

local function installationId()
    local existing = trim(readLocalFile(DEVICE_FILE))
    if #existing >= 20 then
        return existing
    end

    local generated = "install:" .. HttpService:GenerateGUID(false)
    if not writeLocalFile(DEVICE_FILE, generated) then
        error("writefile e necessario para manter o vinculo deste dispositivo")
    end
    return generated
end

local DEVICE_ID = installationId()

local function decodeJson(text)
    local ok, result = pcall(function()
        return HttpService:JSONDecode(text)
    end)
    if ok and type(result) == "table" then
        return result
    end
    return nil
end

local function postJson(path, body)
    local encoded = HttpService:JSONEncode(body)
    local ok, response = pcall(requestFunction, {
        Url = API_BASE .. path,
        Method = "POST",
        Headers = {
            ["content-type"] = "application/json",
            ["accept"] = "application/json, text/plain",
        },
        Body = encoded,
    })

    if not ok then
        return nil, "Falha HTTP: " .. tostring(response)
    end
    if type(response) ~= "table" then
        return nil, "Resposta HTTP desconhecida"
    end

    local status = tonumber(response.StatusCode or response.Status or response.status_code) or 0
    local responseBody = tostring(response.Body or response.body or "")
    return {
        status = status,
        body = responseBody,
        json = decodeJson(responseBody),
    }, nil
end

local function readableApiError(response)
    local code = response and response.json and response.json.error
    return ERROR_MESSAGES[code] or tostring(code or (response and response.body) or "Erro desconhecido")
end

local loading = false
local gui

local function setStatus(text, color)
    if statusLabel then
        statusLabel.Text = tostring(text)
        statusLabel.TextColor3 = color or Color3.fromRGB(190, 194, 205)
    end
end

local function activate(key)
    if loading then
        return false, "Ja existe uma validacao em andamento"
    end
    key = trim(key)
    if key == "" then
        return false, "Digite sua key"
    end

    loading = true
    if activateButton then
        activateButton.Text = "VALIDANDO..."
        activateButton.AutoButtonColor = false
    end
    setStatus("Validando key...", Color3.fromRGB(245, 190, 92))

    local activation, activationError = postJson("/v1/activate", {
        key = key,
        deviceId = DEVICE_ID,
    })

    if not activation then
        loading = false
        if activateButton then
            activateButton.Text = "ATIVAR"
            activateButton.AutoButtonColor = true
        end
        return false, activationError
    end
    if activation.status ~= 200 or not activation.json or activation.json.ok ~= true then
        loading = false
        if activateButton then
            activateButton.Text = "ATIVAR"
            activateButton.AutoButtonColor = true
        end
        return false, readableApiError(activation)
    end

    setStatus("Baixando payload autorizado...", Color3.fromRGB(245, 190, 92))
    local payloadResponse, payloadError = postJson("/v1/payload", {
        token = activation.json.token,
        deviceId = DEVICE_ID,
    })

    if not payloadResponse then
        loading = false
        return false, payloadError
    end
    if payloadResponse.status ~= 200 then
        loading = false
        return false, readableApiError(payloadResponse)
    end

    local compiler = loadstring
    if type(compiler) ~= "function" then
        loading = false
        return false, "loadstring nao esta disponivel"
    end

    local compileOk, chunk = pcall(compiler, payloadResponse.body)
    if not compileOk or type(chunk) ~= "function" then
        loading = false
        return false, "Payload invalido: " .. tostring(chunk)
    end

    writeLocalFile(KEY_FILE, key)
    setStatus("Key aprovada. Iniciando...", Color3.fromRGB(96, 214, 138))
    task.wait(0.2)

    if gui then
        gui:Destroy()
        gui = nil
    end

    local runOk, runError = pcall(chunk)
    loading = false
    if not runOk then
        warn("[KeyLoader] Erro no payload: " .. tostring(runError))
        return false, "Erro ao iniciar payload: " .. tostring(runError)
    end

    return true
end

local function beginActivation()
    task.spawn(function()
        local ok, message = activate(keyBox.Text)
        if not ok then
            setStatus(message, Color3.fromRGB(255, 118, 118))
            if activateButton then
                activateButton.Text = "ATIVAR"
                activateButton.AutoButtonColor = true
            end
        end
    end)
end

activateButton.MouseButton1Click:Connect(beginActivation)

keyBox.FocusLost:Connect(function(enterPressed)
    if enterPressed then
        beginActivation()
    end
end)

-- Se ja existe uma key salva, tenta validar automaticamente.

        local ok, message = activate(keyBox.Text)
        if not ok then

end
